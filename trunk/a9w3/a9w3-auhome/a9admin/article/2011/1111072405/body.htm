================== tty: ssh ======================
# ssh login
# http://sourceforge.net/apps/trac/sourceforge/wiki/Shell%20service
ssh -t trydofor,a9w3@shell.sourceforge.net create

# clear the workspace
cd workspace
rm -rf a9w3

# zip the head version
svn export https://a9w3.svn.sourceforge.net/svnroot/a9w3/trunk/a9w3 a9w3
cd a9w3
zip -r /home/project-web/a9w3/htdocs/a9w3.zip *

# deploy to web
cd /home/project-web/a9w3/htdocs/
unzip -o a9w3.zip

# logout
shutdown
exit
